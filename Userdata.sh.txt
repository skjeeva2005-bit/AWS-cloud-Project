#!/bin/bash

# Update system packages
yum update -y

# Install Apache web server
yum install httpd -y

# Start Apache service
systemctl start httpd

# Enable Apache to start on boot
systemctl enable httpd

# Install additional tools (optional but good practice)
yum install git -y

# Get instance metadata (for dynamic webpage)
INSTANCE_ID=$(curl -s http://169.254.169.254/latest/meta-data/instance-id)
AVAILABILITY_ZONE=$(curl -s http://169.254.169.254/latest/meta-data/placement/availability-zone)

# Create a simple web page
cat <<EOF > /var/www/html/index.html
<!DOCTYPE html>
<html>
<head>
    <title>AWS Cloud Project</title>
    <style>
        body { font-family: Arial; text-align: center; margin-top: 50px; }
        h1 { color: green; }
    </style>
</head>
<body>
    <h1>Secure Multi-Tier Cloud Architecture</h1>
    <p>Web Server is Running Successfully!</p>
    <p><b>Instance ID:</b> $INSTANCE_ID</p>
    <p><b>Availability Zone:</b> $AVAILABILITY_ZONE</p>
</body>
</html>
EOF

# Set proper permissions
chmod 644 /var/www/html/index.html

# Restart Apache service
systemctl restart httpd
